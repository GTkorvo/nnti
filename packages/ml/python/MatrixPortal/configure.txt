# Example of configuration script for MatrixPortal.
# Ideally, you will enable the METIS support for ML (--with-ml_metis);
# and the HDF5 support for EpetraExt (--enable-epetraext-hdf5). The location
# of the header files and the library files for these two libraries
# should be specified in --with-incdirs, --with-ldflags and --with-libs.
#
$HOME/Trilinos/configure \
  CXX=<here CXX compiler> \
  F77=<here fortran compiler> \
  FFLAGS="-O2" \
  --with-incdirs="-I$HOME/include" \
  --with-ldflags="-L$HOME/lib" \
  --with-libs="-lhdf5 -lmetis -llapack -lblas -lz" \
  --prefix=$HOME/Trilinos/LINUX_SERIAL \
  --with-install="/usr/bin/install -p" \
  --with-gnumake \
  --disable-default-packages \
  --disable-examples \
  --disable-tests \
  --enable-python \
  --enable-teuchos \
  --enable-epetra \
  --enable-epetraext \
  --enable-epetraext-hdf5 \
  --enable-triutils \
  --enable-galeri \
  --enable-ifpack \
  --enable-aztecoo \
  --enable-amesos \
  --enable-amesos-klu \
  --enable-ml \
  --with-ml_metis
