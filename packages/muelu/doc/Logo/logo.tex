\newcommand\shadeletter[1]{
  \newcommand\xoffset{0.12}
  \newcommand\yoffset{-0.08}
  % Blur
  \foreach \x in {-0.01,0.01,0.02,0.03} {
    \foreach \y in {-0.03,-0.02,-0.01,0.01} {
      \node[black!60!white,opacity=0.05] at (\xoffset em+\x em,\yoffset em+\y em) {
        \scalebox{2}{{#1}}
      };
    }
  }

  % Main Shadow
  \node[black!100!white,opacity=0.7] at (\xoffset em,\yoffset em) {
    \scalebox{2.1}{{#1}}
  };
  \node at (0,0) {
    \scalebox{2}{\color{white}{#1}}
  };
}
