	"Build Teuchos examples."
	OFF
	)

	"Enable experimental Teuchos features."
	ON
	)

	"Enable Teuchos features for complex arithmetic."
	ON
	)

ENDIF(TEUCHOS_ENABLE_TESTS)

	ADD_SUBDIRECTORY(example)
ENDIF(TEUCHOS_ENABLE_EXAMPLES)
